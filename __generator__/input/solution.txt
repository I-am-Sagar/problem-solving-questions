#include<bits/stdc++.h>
using namespace std;
#define ll long long

ll min(ll x, ll y) 
{ 
  return (x < y) ? x : y ;
}

int main()
{
    ll n, X, Y;
    cin >> n >> X >> Y;
    
    string s;
    cin >> s;
    
    ll zero_groups = 0;
    if (s[0] == '0')
        zero_groups++;

    
    for (ll i = 1; i <n; i++) {
        if (s[i] == '0' && s[i-1] == '1')
            zero_groups++;
    }
    ll cost;
    
    if(zero_groups == 0)
        cost = 0;
    else
        cost = (zero_groups-1)*min(X, Y) + Y;
    
    cout << cost << endl;
    
	return 0;
}
